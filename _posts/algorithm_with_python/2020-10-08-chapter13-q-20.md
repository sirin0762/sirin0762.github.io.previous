n = int(input())

maps = [list(input().split()) for _ in range(n)]
temp = [[0] * n for _ in range(n)]

dx, dy = [-1, 1, 0, 0], [0, 0, -1, 1]

result = []

def check_stu(x, y, direction):
  nx, ny = x + dx[direction], y + dy[direction]
  if nx < n and ny < n and nx >= 0 and ny >= 0:
    if temp[nx][ny] == 'O' or temp[nx][ny] == 'T':
      return
    elif temp[nx][ny] == 'S':
      temp[nx][ny] = 'X'
      check_stu(nx, ny, direction)

def dfs(count):
  if count == 3:
    for i in range(n):
      for j in range(n):
        temp[i][j] = maps[i][j]

    for i in range(n):
      for j in range(n):
        if temp[i][j] == 'T':
          for direction in range(4):
            check_stu(i, j, direction)

    for i in range(n):
      for j in range(n):
        if temp[i][j] == 'S':
          return False

  for i in range(n):
    for j in range(n):
      if maps[i][j] == 'X':
        count += 1
        maps[i][j] = 'O'
        dfs(count)
        maps[i][j] = 'X'
        count -= 1
  print(temp)
  return True

print(dfs(0))
